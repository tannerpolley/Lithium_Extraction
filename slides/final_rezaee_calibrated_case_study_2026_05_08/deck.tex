\section{Case Basis}

\begin{frame}{Why Smackover Is The Flagship Feed}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \includegraphics[width=\linewidth]{brine_li_tds_scatter.png}
    \end{column}
    \begin{column}{0.41\textwidth}
      \begin{alertblock}{Engineering premise}
        Southern Arkansas Smackover brine couples high lithium value with
        hypersaline electrolyte severity.
      \end{alertblock}
      \begin{itemize}
        \item Selected clean-feed basis: \SI{168}{mg/L} Li and
          \SI{64100}{mg/L} Na.
        \item TDS is retained as a severity variable.
        \item Raw divalents are pretreatment context, not active solvent-model
          competitors.
      \end{itemize}
      \sourcefoot{Sources: USGS Smackover data release; local screening table.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Pretreatment Defines The Extraction Feed}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \includegraphics[width=\linewidth]{rezaee_surrogate_input_ranges.png}
    \end{column}
    \begin{column}{0.43\textwidth}
      \processbridge{0.86}
      \vspace{0.2cm}
      \begin{itemize}
        \item The solvent model sees a clean Li/Na stream.
        \item \ce{Ca^{2+}}, \ce{Mg^{2+}}, \ce{Sr^{2+}}, and \ce{Ba^{2+}} are
          removed upstream.
        \item Produced-water severity is carried through TDS and Na/Li flags.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Thermodynamic Bridge}

\begin{frame}{Rezaee DES/TOPO Becomes A Transfer-Variable Bridge}
  \slidetag{THERMODYNAMIC SURROGATE CONTRACT}
  \begin{center}
    \includegraphics[width=0.82\linewidth]{rezaee_surrogate_pipeline.png}
  \end{center}
  \begin{itemize}
    \item Rezaee extraction responses are converted to Li and Na distribution
      coefficients.
    \item Calibrated $\log D_i$ surfaces generate the produced-water screening
      matrix.
    \item Model-basis and validity flags are carried into downstream process
      tables.
  \end{itemize}
  \sourcefoot{Sources: Rezaee et al. 2025 extraction data; Rezaee et al. 2026 parameter and reaction-equilibrium support.}
\end{frame}

\begin{frame}{Distribution Coefficients Are The Process Contract}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \[
        D_i = \frac{E_i}{100-E_i}\frac{1}{O/A}
      \]
      \[
        \SLiNa = \frac{\DLi}{\DNa}
      \]
      \begin{exampleblock}{Nominal MS-2, $O/A=1$}
        $E_{\mathrm{Li}}=50.66\%$, $E_{\mathrm{Na}}=5.67\%$,
        $\SLiNa=17.09$.
      \end{exampleblock}
    \end{column}
    \begin{column}{0.49\textwidth}
      \includegraphics[width=\linewidth]{nominal_transfer_math_card.png}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Lithium Extraction Surface}
  \begin{columns}[T,onlytextwidth]

\begin{frame}{Validation Boundary}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \validationriskplot
    \end{column}
    \begin{column}{0.40\textwidth}
      \begin{alertblock}{Boundary condition}
        Every produced-water row is flagged as outside the original Rezaee Na/Li
        DOE training domain.
      \end{alertblock}
      \begin{itemize}
        \item Process screening is appropriate.
        \item Final physical acceptance requires high-Na validation.
        \item Direct reactive-LLE closure is not claimed in the main result.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[standout]
The case study is ready for PrOMMiS/IDAES integration when $D_i$,
$\SLiNa$, and validity flags remain traceable from the Rezaee basis.
\end{frame}

\begin{frame}{Final Technical Message}
  \begin{enumerate}
    \item Use Rezaee DES/TOPO as the active Li/Na case-study bridge.
    \item Present Smackover as the produced-water source basis and pretreatment
      context.
    \item Move only traceable transfer variables into PrOMMiS/IDAES.
    \item Keep archived HBTA/TOPO, Gando, Jang, Yu, and Hubach material outside
      the main chain unless explicitly labeled as comparison or diagnostics.
  \end{enumerate}
  \begin{block}{Next rigorous step}
    Replace the algebraic process handoff with a full PrOMMiS MSContactor solve
    and an IDAES costing block once the final surrogate run interface is fixed.
  \end{block}
\end{frame}
